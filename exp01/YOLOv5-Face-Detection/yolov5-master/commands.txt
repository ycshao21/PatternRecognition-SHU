#训练:
#--img 输入图片尺寸
#--batch 输入批大小
#--epochs 训练周期数
#--weights 初始权重 若不使用预训练权重则设为 --weights '' --cfg yolov5s.yaml
#--data 数据配置信息
#--device 运行设备 i.e. 0 or 0,1,2,3 or cpu
#其他命令行参数可查看train.py中433行开始的parse_opt函数
#需要根据自己设备情况调整img与batch参数的大小
#一般显存越大能使用的batch上限越大
#无cuda显卡则使用cpu训练

#在WIDER_FACE_smalls上训练
python train.py  --img 640 --batch 30 --epochs 100 --data WIDER_FACE_small/WIDER_FACE_small.yaml --weights yolov5s.pt --nosave --cache
#在WIDER_FACE_tiny上训练
python train.py  --img 640 --batch 30 --epochs 100 --data WIDER_FACE_tiny/WIDER_FACE_tiny.yaml --weights yolov5s.pt --nosave --cache
